1、插件介绍
  1. 相关文章： http://qianduanblog.com/2595.html
  2. 解析和操作URL的search部分，默认是不会跳转的。
2、插件API
// 1. 解析

$.search();

// =>{...}



// 2. 获取

$.search("a");

// =>"1"

$.search(["a","b"]);

// =>{a:1,b:2}



// 3.1 设置

$.search("a",1);

$.search({a:1,b:2});



// 3.2 设置并跳转

$.search("a",1,true);

$.search({a:1,b:2},true);



// 4.1 删除

$.search("a",null);

// =>"b=1"

$.search(["a","b"],null);

// =>""



// 4.2 删除并跳转

$.search("a",null,true);

$.search(["a","b"],null,true);



// 5.1 清空

$.search(null);



// 5.2 清空并跳转

$.search(null,true);
