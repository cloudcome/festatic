1、插件说明
  1. 插件支持localStorge和sessionStorage
2、插件API

2.1、options

// options
$.storage.defaults = {
    // 默认为 localStorage，可选 session为sessionStorage
    type: 'local'
};
$.storage({
    // 默认为 localStorage，可选 session为sessionStorage
    type: 'local'
});

2.2、set

// set
$.storage().set('key', 'val');
$.storage().set({
	key1: 'val1',
	key2: 'val2',
});

2.3、get

// get
$.storage().get('key');
$.storage().get(['key1', 'key2']);

2.4、remove

// remove
$.storage().remove('key');
$.storage().remove(['key1', 'key2']);

2.5、clear

// clear
$.storage().clear();

2.6、bind

// bind
$.storage().bind('key', fn);

2.7、unbind

// unbind
$.storage().unbind('key', fn);
// unbind all
$.storage().unbind('key');
